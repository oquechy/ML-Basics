def read_dataset(dataset, to_labels=None, label_col='label'):
    data = pd.read_csv(dataset)
    Y = data[label_col].to_numpy()
    X = data.drop(label_col, axis=1).to_numpy()
    if to_labels:
        Y = np.array([to_labels[y] for y in Y])
    return X, Y

def r2(X, y, w):
    u = norm(X.dot(w) - y)
    v = norm(-y + np.average(y))
    return 1 - u / v

def plot(X, X_poly, w):
    plt.cla()
    plt.scatter(X, y)
    plt.plot(X, X_poly.dot(w), 'r')
    plt.show()
    
def regression(X, y, deg=1, show=False):
    features = PolynomialFeatures(deg)
    X_poly = features.fit_transform(X)
    train, test, labels_train, labels_test = train_test_split(X_poly, y, test_size=0.20)
    w = inv(train.transpose() @ train) @ train.transpose() @ labels_train
    print(f'R^2 train {r2(train, labels_train, w)}')
    print(f'R^2 test  {r2(test, labels_test, w)}')
    if show:
        plot(X, X_poly, w)

def ridge_regression(X, y, a, deg=1, show=False):
    features = PolynomialFeatures(deg)
    X_poly = features.fit_transform(X)
    train, test, labels_train, labels_test = train_test_split(X_poly, y, test_size=0.20)
    w = inv(train.transpose() @ train + np.eye(train.shape[1])) @ train.transpose() @ labels_train
    if show:
        plot(X, X_poly, w)
    return r2(train, labels_train, w), r2(test, labels_test, w)

from sklearn.linear_model import Lasso

def lasso(X, y, a=1e-3, max_iter=10e5, deg=1, draw=True):
    features = PolynomialFeatures(deg)
    X_poly = features.fit_transform(X)
    train, test, labels_train, labels_test = train_test_split(X_poly, y, test_size=0.20)
    lasso = Lasso(alpha=a, max_iter=max_iter)
    lasso.fit(train, labels_train)
    print(f'Deg {deg}')
    print(f'R^2 train {lasso.score(train, labels_train)}')
    print(f'R^2 test  {lasso.score(test, labels_test)}')
    print(f'{np.sum(lasso.coef_ != 0)} features used')
    if draw:
        plot(X, X_poly, lasso.coef_)

def read_path(dataset):
    data = pd.read_csv(dataset)
    X = data.to_numpy()[:,1:].astype('float')
    return X

def dist(X):
    return np.sum(abs(X[1:] - X[:-1]))
    
def plot_manhattan(X):
    path = []
    for i, (x, y) in enumerate(X[:-1]):
        path.append(np.array([x, y]))
        path.append(np.array([X[i + 1][0], y]))
        path.append(np.array([X[i + 1][0], X[i + 1][1]]))
    path = np.array(path)
    plt.cla()
    plt.plot(path[:, 0], path[:, 1])
    plt.scatter(X[:, 0], X[:, 1])
    plt.show()
    
def monte_carlo(X, iterations=10000):
    opt_path = X
    opt_dist = dist(X)
    for i in range(iterations):
        shuffle(X)
        d = dist(X)
        if d < opt_dist:
            opt_dist = d
            opt_path = np.copy(X)
    return opt_path, opt_dist

def random_walk(X, iterations=100000):
    opt_path = shuffle(X)
    opt_dist = dist(X)
    for i in range(iterations):
        l, r = randint(0, X.shape[0], 2)
        if l > r:
            l, r = r, l
        X[l:r + 1] = X[l:r + 1][::-1]
        d = dist(X)
        if d < opt_dist:
            opt_dist = d
            opt_path = np.copy(X)
    return opt_path, opt_dist

def hill_climb(X, iterations=100000):
    opt_path = shuffle(X)
    opt_dist = dist(X)
    for i in range(iterations):
        l, r = randint(0, X.shape[0], 2)
        if l > r:
            l, r = r, l
        X[l:r + 1] = X[l:r + 1][::-1]
        d = dist(X)
        if d < opt_dist:
            opt_dist = d
            opt_path = np.copy(X)
        else:
            X[l:r + 1] = X[l:r + 1][::-1] 
    return opt_path, opt_dist

def annealing(X, temp_high=50, temp_low=1, iterations=100000):
    opt_path = shuffle(X)
    opt_dist = dist(X)
    for t in np.linspace(temp_high, temp_low, iterations):
        l, r = randint(0, X.shape[0], 2)
        if l > r:
            l, r = r, l
        X[l:r + 1] = X[l:r + 1][::-1]
        d = dist(X)
        if d < opt_dist:
            opt_dist = d
            opt_path = np.copy(X)
        elif np.random.rand() > np.exp((opt_dist - d) / t):
            X[l:r + 1] = X[l:r + 1][::-1]
    return opt_path, opt_dist

def crossing_over(population):
    idx = randint(0, population.shape[0], 2)
    x, y = population[idx]
    l, r = randint(0, x.shape[0], 2)
    if l > r:
        l, r = r, l
    repl = x[l:r]
    base = np.array([point for point in y if point not in repl])
    z = np.concatenate((base[:l], repl, base[l:]))
    return z

def genetic(X, generations=1000, population_size=10, offspring_rate=2):
    opt_path = X
    opt_dist = dist(X)
    population = np.array([X.copy() for _ in range(population_size)])
    for gen in range(generations):
        offspring = [crossing_over(population) for _ in range(population_size * offspring_rate)]
        dists = np.array([dist(specimen) for specimen in offspring])
        i = np.argmin(dists)
        if dists[i] < opt_dist:
            opt_dist = dists[i]
            opt_path = offspring[i]
        idx = np.argpartition(dists, population_size)[:population_size]
        population = np.array(offspring)[idx]
    return opt_path, opt_dist
