def read_dataset(dataset, to_labels=None):
    data = pd.read_csv(dataset)
    Y = data['label'].to_numpy()
    X = data.drop('label', axis=1).to_numpy()
    if to_labels:
        Y = np.array([to_labels[y] for y in Y])
    return X, Y

def plot_decision_function(model, xlim, ylim):
    x = np.linspace(xlim[0], xlim[1], 30)
    y = np.linspace(ylim[0], ylim[1], 30)
    Y, X = np.meshgrid(y, x)
    xy = np.vstack([X.ravel(), Y.ravel()]).T
    P = model.decision_function(xy).reshape(X.shape)
    
    plt.contour(X, Y, P, colors='k',
               levels=[-1, 0, 1],
               linestyles=['--', '-', '--'])
    
    plt.scatter(model.support_vectors_[:, 0],
                model.support_vectors_[:, 1], facecolors='none', edgecolor='black', linewidth=1.5);
    

def plot_svm(X, Y, kernel, degree):
    model = SVC(gamma='auto', kernel=kernel, degree=degree)
    model.fit(X, Y)
    
    plt.scatter(X[Y == 1][:, 0], X[Y == 1][:, 1])
    plt.scatter(X[Y == -1][:, 0], X[Y == -1][:, 1])
    plot_decision_function(model, [np.min(X[:, 0]), np.max(X[:, 0])], [np.min(X[:, 1]), np.max(X[:, 1])])
    

for kernel, degree in [('linear', 0), ('poly', 2), ('poly', 3), ('poly', 5), ('rbf', 0)]:
    plt.cla()
    plot_svm(X, Y, kernel, degree)
    plt.show()

from time import time


def compare_times(train, test, labels_train, labels_test):
    svm_model = SVC(gamma='auto', kernel='poly', degree=1)
    time1 = time()
    svm_model.fit(train, labels_train)
    time2 = time()
    print(f'svn fit took {((time2-time1)*1000):.2f} ms')
    svm_accuracy = svm_model.score(test, labels_test)

    print(f'svm accuracy: {svm_accuracy}')

    rf_model = RandomForestClassifier(n_estimators=5, max_depth=3)
    time1 = time()
    rf_model.fit(train, labels_train)
    time2 = time()
    print(f'rf fit took {((time2-time1)*1000):.2f} ms')
    rf_accuracy = accuracy_score(rf_model.predict(test), labels_test)

    print(f'rf accuracy: {rf_accuracy}')

from scipy.stats import norm
from math import isnan


def nb_classify_gaussian(test, train, labels_train):
    prob = np.ones((test.shape[0], 2))
    for l, count in enumerate(np.bincount(labels_train)):
        prob[:, l] *= count / labels_train.shape[0]
        
        for f in tqdm_notebook(range(train.shape[1])):
            feature = train[labels_train == l][:, f]
            mean = np.mean(feature)
            std = np.std(feature)
            for i, x in enumerate(test):
                prob[i][l] *= norm.pdf(x[f], loc=mean, scale=std)
    
    assert all(all(not isnan(val) for val in row) for row in prob)
    return prob
    
    
def normalize(scores):
    return np.nan_to_num(scores / (scores[:, 0] + scores[:, 1])[:, None])


def plot_roc(scores, labels_test):
    scores = normalize(scores)
    fpr, tpr, thresholds = roc_curve(labels_test, scores[:, 1], pos_label=1)
    auc = roc_auc_score(labels_test, scores[:, 1])
    plt.plot(fpr, tpr)
    print(f'auc {auc}')

from nltk.stem import LancasterStemmer
from nltk.tokenize import RegexpTokenizer

from collections import Counter


def parse(X, tokenizer=RegexpTokenizer(r'\w+'), stemmer=LancasterStemmer()):
    return np.array([{stemmer.stem(word) for word in tokenizer.tokenize(sms)} for [sms] in X])
    

def dicts(train, labels_train, lim):
    ham, spam = Counter(), Counter()
    for train, counter in [(train[labels_train == 0], ham), (train[labels_train == 1], spam)]:
        for sms in train:
            for word in sms:
                counter[word] += 1
    dictionary = {word for word, _ in ham.most_common(lim) + spam.most_common(lim)}
    return {word: ham[word] for word in dictionary}, {word: spam[word] for word in dictionary}, dictionary


def nb_classify_discrete(test, train, labels_train, dict_size=120):
    ham, spam, dictionary = dicts(train, labels_train, dict_size)

    prob = np.ones((test.shape[0], 2))
    for l, count in enumerate(np.bincount(labels_train)):
        label_dict = [ham, spam][l]
        prob[:, l] *= count / labels_train.shape[0]
        
        for word in tqdm_notebook(dictionary):
            for i, sms in enumerate(test):
                same_feature = label_dict[word] if word in sms else count - label_dict[word]
                prob[i][l] *= (same_feature + 1) / (count + 2)
    
    assert all(all(not isnan(val) for val in row) for row in prob)
    return prob
    

def normalize(scores):
    return np.nan_to_num(scores / (scores[:, 0] + scores[:, 1])[:, None])


def plot_roc(scores, labels_test):
    scores = normalize(scores)
    fpr, tpr, thresholds = roc_curve(labels_test, scores[:, 1], pos_label=1)
    auc = roc_auc_score(labels_test, scores[:, 1])
    plt.plot(fpr, tpr)
    print(f'auc {auc}')
