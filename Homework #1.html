def read_dataset(dataset, to_labels=None):
    data = pd.read_csv(dataset)
    Y = data['label'].to_numpy()
    X = data.drop('label', axis=1).to_numpy()
    if to_labels:
        Y = np.array([to_labels[y] for y in Y])
    return X, Y


def scale(X):
    scaler = MinMaxScaler()
    scaler.fit(X)
    return scaler.transform(X)
    
    
def knn_neighbors(x, X, K=10):
    norms = [norm(row) for row in X - x]
    nearest = np.argsort(norms)[:K]
    return nearest


def knn_labels(x, X, Y, K=10):
    nearest = knn_neighbors(x, X, K)
    labels = [Y[i] for i in nearest]
    counts = [np.bincount(labels[:i]) for i in range(1, K + 1)]
    return [np.argmax(row) for row in counts]


def loo(X, Y, K=10):
    error = np.zeros(K)
    for i, row in tqdm_notebook(enumerate(X)):
        labels = knn_labels(row, np.delete(X, i, axis=0), np.delete(Y, i), K)
        error += [0 if label == Y[i] else 1 for label in labels]
    return error / X.shape[0]

def dbscan(X, e, m):
    neighbors = []
    for x in X:
        neighborhood = [i for i, row in enumerate(X) if norm(row - x) <= e]
        neighbors.append(neighborhood)
    
    clusters = np.zeros(X.shape[0])
    def dfs(v, cluster):
        if clusters[v] or len(neighbors[v]) <= m:
            return False
        clusters[v] = cluster
        for u in neighbors[v]:
            dfs(u, cluster)
        return True
    
    cluster = 1
    for i in range(X.shape[0]):
        if dfs(i, cluster):
            cluster += 1
    
    for i, neighborhood in enumerate(neighbors):
        for n in neighborhood:
            if not clusters[n]:
                clusters[n] = clusters[i]
    return cluster - 1, clusters

def agglomerative_clustering(X, counts, metric):
    result = []
    clusters = [[i] for i in range(X.shape[0])]
    
    for count in tqdm_notebook(range(len(clusters), 1, -1)):
        if count in counts:
            result.append(clusters.copy())
        closest = (0, 1)
        min_metric = metric(X, clusters[0], clusters[1])
        for i in range(len(clusters)):
            for j in range(i + 1, len(clusters)):
                cur_metric = metric(X, clusters[i], clusters[j])
                if cur_metric < min_metric:
                    min_metric = cur_metric
                    closest = (i, j)
        clusters[closest[0]].extend(clusters[closest[1]])
        clusters.pop(closest[1])
    return result

def average_dist_metric(X, a, b):
    cluster = X[np.concatenate([a, b])]
    return np.sum(pdist(cluster)) * 2 / cluster.shape[0] / (cluster.shape[0] - 1)

def k_mean(X, count, e=1e-5):
    centers = X[np.random.choice(range(X.shape[0]), count, False)]
    while True:
        closest_center = np.array([np.argmin(norm(centers - x, axis=1)) for x in X])
        new_centers = []
        for cluster in range(count):
            points = [i for i, c in enumerate(closest_center) if c == cluster]
            new_centers.append(np.sum(X[points], axis=0) / len(points))
        new_centers = np.array(new_centers)
        if norm(new_centers - centers) < e:
            return len(centers), closest_center
        centers = new_centers

def purity(count, clusters, Y):
    metric = 0
    for cluster in range(count):
        labels = [Y[i] for i, c in enumerate(clusters) if c == cluster]
        label_frequencies = np.bincount(labels)
        metric += np.max(label_frequencies)
    return metric / clusters.shape[0]

def roc(feature, label, e=1e-4):
    tp, tn, fp, fn = [], [], [], []
    for threshold in sorted(np.unique(feature)):# np.arange(0, 1, e):
        hypothesis = feature < threshold
        tp.append(np.sum((1 == label) & (hypothesis == label)))        
        tn.append(np.sum((0 == label) & (hypothesis == label)))        
        fp.append(np.sum((0 == label) & (hypothesis != label)))
        fn.append(np.sum((1 == label) & (hypothesis != label)))
    tp /= np.sum(Y == 1)
    fp /= np.sum(Y == 0)
    fn /= np.sum(Y == 1)
    tn /= np.sum(Y == 0)
    auc = np.trapz(tp, fp)
    return (auc, tp, fp) if auc >= 0.5 else (1 - auc, fn, tn)
    

def plot_roc_curves(X, Y, n):
    rocs = []
    for feature in tqdm_notebook(range(X.shape[1])):
        auc, tp, fp = roc(X[:, feature], Y)
        rocs.append((auc, tp, fp, feature))

    for auc, tp, fp, feature in sorted(rocs)[-n:]:
        plt.plot(fp, tp)
        print(f'auc: {auc}\tfeature: {feature}')

def split_train_validation(X, validation_proportion):
    mask = np.random.random_sample(X.shape[0]) < validation_proportion
    return ~mask, mask


def class_proportions(data, Y):
    labels = Y[data]
    return np.bincount(labels) / labels.shape[0]

def IG(hyp, Y):
    return hyp.shape[0] * I(Y) - np.sum(hyp) * I(Y[hyp]) - np.sum(~hyp) * I(Y[~hyp])
    

def I(Y):
    if Y.shape[0] == 0:
        return 0
    return np.sum(Y) * np.sum(~Y) / Y.shape[0] / Y.shape[0]


class Node():
    def __init__(self, feature, threshold, left=None, right=None, class_prob=None):
        self.threshold = threshold
        self.feature = feature
        self.left = left
        self.right = right
        self.class_prob = class_prob
    
    def __str__(self):
        return f'Threshold {self.threshold}\nFeature {self.feature}\nLeft [{self.left}]\nRight [{self.right}]\n'


class DecisionTree():
    def __init__(self, X, Y, depth=1):
        self.root = self.build_node(X, Y, depth)
        
    def build_node(self, X, Y, depth, e=0.001):
        if not depth:
            return None
        
        opt_feature, opt_threshold, opt_ig, opt_hyp = 0, 0, 0, np.array([False * Y.shape[0]])
        for f in range(X.shape[1]):
            feature = X[:, f]
            for threshold in np.unique(feature):
                hypothesis = feature <= threshold
                ig = IG(hypothesis, Y)
                if opt_ig < ig:
                    opt_ig = ig
                    opt_hyp = hypothesis.copy()
                    opt_feature = f
                    opt_threshold = threshold

        if depth == 1 or opt_ig <= 0:
            counts = np.bincount(Y, minlength=2) / Y.shape[0]
            return Node(opt_feature, opt_threshold, class_prob=counts)
        
        left = self.build_node(X[opt_hyp], Y[opt_hyp], depth - 1) 
        right = self.build_node(X[~opt_hyp], Y[~opt_hyp], depth - 1) 
        return Node(opt_feature, opt_threshold, left, right)
    
    def classify_rec(self, node, x):
        if node.class_prob is not None:
            return node.class_prob
        return self.classify_rec(node.left if x[node.feature] <= node.threshold else node.right, x)
    
    def classify(self, x):
        return self.classify_rec(self.root, x)
    
    
def get_opt_depth(train, validation):
    opt_auc, opt_depth = 0, 1
    for depth in tqdm_notebook(range(4, 11)):
        tree = DecisionTree(X[train], Y[train], depth)
        scores = np.array([])
        scores = np.array([tree.classify(x)[1] for x in X[validation]])
        fpr, tpr, thresholds = roc_curve(Y[validation], scores, pos_label=1)
        auc = roc_auc_score(Y[validation], scores)
        plt.plot(fpr, tpr)
        if auc > opt_auc:
            opt_auc = auc
            opt_depth = depth
    return opt_auc, opt_depth
